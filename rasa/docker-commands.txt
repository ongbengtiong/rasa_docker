  -v ${pwd}/:/app `
  --network "host" `
  rasa/rasa:latest-full `

    --config config.yml `

    --data data `
    --out models

  run `
    python -m rasa_nlu.train `
    -c config.yml `
    -d project/data/nlu.md `
    -o models `
    --project current
