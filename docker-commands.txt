# powershell commands

# change the ${project} to the folder of the project. The project folder should under the root folder of rasa_docker/
${project} = 'meeting'

docker run `
  -v ${pwd}/projects/${project}:/app `
  rasa/rasa:latest-full `
  train   `
    --config config.yml `
    --domain domain.yml `
    --data data `
    --out models



docker run `
  -v ${pwd}/projects/${project}/models:/app/models `
  -p 5005:5005
  rasa/rasa:latest-full `
  run



docker run `
  -v ${pwd}/projects/${project}:/app/project `
  -v ${pwd}/projects/${project}/models/rasa_nlu:/app/models `
  rasa/rasa_nlu:latest-spacy `
  run `
    python -m rasa_nlu.train `
    -c config.yml `
    -d project/data/nlu.md `
    -o models `
    --project current


	
docker run `
  -it `
  -v ${pwd}/projects/${project}/models/rasa_core:/app/models `
  rasa/rasa_core:latest `
  start `
    --core models
	

docker run `
  -v ${pwd}/projects/${project}:/app/project `
  -v ${pwd}/projects/${project}/models/rasa_nlu:/app/models `
  rasa/rasa_nlu:latest-spacy `
  run `
    python -m rasa_nlu.train `
    -c config.yml `
    -d project/data/nlu.md `
    -o models `
    --project current
